document.getElementById("search-form").addEventListener("submit", async (event) => { event.preventDefault(); const query = document.getElementById("query").value; const resultsDiv = document.getElementById("results"); const downloadButton = document.getElementById("download-results"); try { // Add loading indicator resultsDiv.innerHTML = '

Načítání...

'; // Use full URL if running on different port/domain const response = await fetch(`http://localhost:5000/search?query=${encodeURIComponent(query)}`, { method: 'GET', headers: { 'Accept': 'application/json' } }); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } const results = await response.json(); // Clear previous results resultsDiv.innerHTML = `

Výsledky:

`; // Handle no results if (results.length === 0) { resultsDiv.innerHTML += '

Nebyly nalezeny žádné výsledky.

'; downloadButton.style.display = "none"; return; } // Display results with error handling results.forEach(result => { if (result.title && result.link) { resultsDiv.innerHTML += `

${result.title}

${result.link}
`; } }); // Show download button downloadButton.style.display = results.length > 0 ? "inline-block" : "none"; } catch (error) { console.error('Search error:', error); resultsDiv.innerHTML = `

Chyba při hledání: ${error.message}

`; downloadButton.style.display = "none"; } });
